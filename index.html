// 1. ConnectionFactory (Pooling)
class ConnectionFactory {
  static final List _pool = ['Conn-A', 'Conn-B'];
  static String getConnection() => _pool.removeAt(0);
  static void releaseConnection(String conn) => _pool.add(conn);
}

// 2. NotificationFactory (Platform Based)
abstract class Notification { void send(); }
class EmailNotification implements Notification { void send() => print("Email"); }
class PushNotification implements Notification { void send() => print("Push"); }
class NotificationFactory {
  static Notification create(String type) => type == 'email' ? EmailNotification() : PushNotification();
}

// 3. ShapeFactory (Caching)
class ShapeFactory {
  static final Map _cache = {};
  static String getShape(String type) {
    return _cache.putIfAbsent(type, () => "Shape: $type");
  }
}

void main() {
  // Demo implementasi
  print(ConnectionFactory.getConnection());
  NotificationFactory.create('email').send();
  print(ShapeFactory.getShape('Circle'));
}
